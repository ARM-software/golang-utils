// dataSource is an interface that returns object which can be read and closed.
type dataSource interface {
	ReadCloser() (io.ReadCloser, error)
}

// sourceFile represents an object that contains content on the local file system.
type sourceFile struct {
	name string
}

func ( sourceFile) () ( io.ReadCloser,  error) {
	return os.Open(.name)
}

// sourceData represents an object that contains content in memory.
type sourceData struct {
	data []byte
}

func ( *sourceData) () (io.ReadCloser, error) {
	return ioutil.NopCloser(bytes.NewReader(.data)), nil
}

// sourceReadCloser represents an input stream with Close method.
type sourceReadCloser struct {
	reader io.ReadCloser
}

func ( *sourceReadCloser) () (io.ReadCloser, error) {
	return .reader, nil
}

func ( interface{}) (dataSource, error) {
	switch s := .(type) {
	case string:
		return sourceFile{}, nil
	case []byte:
		return &sourceData{}, nil
	case io.ReadCloser:
		return &sourceReadCloser{}, nil
	case io.Reader:
		return &sourceReadCloser{ioutil.NopCloser()}, nil
	default:
		return nil, fmt.Errorf("error parsing data source: unknown type %q", )
	}
}
