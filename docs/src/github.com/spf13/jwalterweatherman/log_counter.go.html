// Counter is an io.Writer that increments a counter on Write.
type Counter struct {
	count uint64
}

func ( *Counter) () {
	atomic.AddUint64(&.count, 1)
}

// Reset resets the counter.
func ( *Counter) () {
	atomic.StoreUint64(&.count, 0)
}

// Count returns the current count.
func ( *Counter) () uint64 {
	return atomic.LoadUint64(&.count)
}
