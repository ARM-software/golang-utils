package mem

type Dir interface {
	Len() int
	Names() []string
	Files() []*FileData
	Add(*FileData)
	Remove(*FileData)
}

func ( *FileData,  *FileData) {
	.memDir.Remove()
}

func ( *FileData,  *FileData) {
	.memDir.Add()
}

func ( *FileData) {
	if .memDir == nil {
		.dir = true
		.memDir = &DirMap{}
	}
}
