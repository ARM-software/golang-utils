	if  == nil {
		return nil
	}
	if  := reflect.TypeOf(); .Kind() != reflect.Ptr {
		// Avoid creating a reflect.Value if it's not a pointer.
		return

	 := reflect.ValueOf()
	for .Kind() == reflect.Ptr && !.IsNil() {
		 = .Elem()
	}
	return .Interface()
}
