type bmixer interface {
	bmix(p []byte) (tail []byte)
	Size() (n int)
	reset()
}

	clen int      // Digested input cumulative length.
	tail []byte   // 0 to Size()-1 bytes view of `buf'.
	buf  [16]byte // Expected (but not required) to be Size() large.
