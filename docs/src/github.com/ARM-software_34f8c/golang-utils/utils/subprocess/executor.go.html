// Stops the process if currently working
func ( *Subprocess) () ( error) {
	if !.IsOn() {
		return
	}
	 = .Check()
	if  != nil {
		return
	}
	.mu.Lock()
	defer .mu.Unlock()
	defer .Cancel()
	if !.IsOn() {
		return
	}
	.loggers.Log(fmt.Sprintf("Stopping process [%v]", .subprocess.Pid))
	_ = .subprocess.Kill()
	_ = .command.Wait()
	.command.Process = nil
	.subprocess = nil
	.isRunning.Store(false)
	return
}
