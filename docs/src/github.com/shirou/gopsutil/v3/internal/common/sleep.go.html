// Sleep awaits for provided interval.
// Can be interrupted by context cancelation.
func ( context.Context,  time.Duration) error {
	var  = time.NewTimer()
	select {
	case <-.Done():
		return .Err()
	case <-.C:
		return nil
	}
}
