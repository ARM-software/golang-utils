	if .InodesTotal == 0 {
		.InodesUsedPercent = 0
	} else {
		.InodesUsedPercent = (float64(.InodesUsed) / float64(.InodesTotal)) * 100.0
	}

	if (.Used + .Free) == 0 {
		.UsedPercent = 0
	} else {
		// We don't use ret.Total to calculate percent.
		// see https://github.com/shirou/gopsutil/issues/562
		.UsedPercent = (float64(.Used) / float64(.Used+.Free)) * 100.0
	}

	return , nil
}
