var (
	// uint8Type is a reflect.Type representing a uint8.  It is used to
	// convert cgo types to uint8 slices for hexdumping.
	uint8Type = reflect.TypeOf(uint8(0))

	// cCharRE is a regular expression that matches a cgo char.
	// It is used to detect character arrays to hexdump them.
	cCharRE = regexp.MustCompile(`^.*\._Ctype_char$`)

	// cUnsignedCharRE is a regular expression that matches a cgo unsigned
	// char.  It is used to detect unsigned character arrays to hexdump
	// them.
	cUnsignedCharRE = regexp.MustCompile(`^.*\._Ctype_unsignedchar$`)

	// cUint8tCharRE is a regular expression that matches a cgo uint8_t.
	// It is used to detect uint8_t arrays to hexdump them.
	cUint8tCharRE = regexp.MustCompile(`^.*\._Ctype_uint8_t$`)
)

// dumpState contains information about the state of a dump operation.

// indent performs indentation according to the depth level and cs.Indent
// option.
func ( *dumpState) () {
	if .ignoreNextIndent {
		.ignoreNextIndent = false
		return
	}
	.w.Write(bytes.Repeat([]byte(.cs.Indent), .depth))
}

// dump is the main workhorse for dumping a value.  It uses the passed reflect
// value to figure out what kind of object we are dealing with and formats it
// appropriately.  It is a recursive function, however circular data structures
// are detected and handled properly.
func ( *dumpState) ( reflect.Value) {
	// Handle invalid reflect values immediately.
	 := .Kind()
	if  == reflect.Invalid {
		.w.Write(invalidAngleBytes)
		return
	}

	// Handle pointers specially.
	if  == reflect.Ptr {
		.indent()
		.dumpPtr()
		return
	}

	// Print type information unless already handled elsewhere.
	if !.ignoreNextType {
		.indent()
		.w.Write(openParenBytes)
		.w.Write([]byte(.Type().String()))
		.w.Write(closeParenBytes)
		.w.Write(spaceBytes)
	}
	.ignoreNextType = false

	// Display length and capacity if the built-in len and cap functions
	// work with the value's kind and the len/cap itself is non-zero.
	,  := 0, 0
	switch .Kind() {
	case reflect.Array, reflect.Slice, reflect.Chan:
		,  = .Len(), .Cap()
	case reflect.Map, reflect.String:
		 = .Len()
	}
	if  != 0 || !.cs.DisableCapacities &&  != 0 {
		.w.Write(openParenBytes)
		if  != 0 {
			.w.Write(lenEqualsBytes)
			printInt(.w, int64(), 10)
		}
		if !.cs.DisableCapacities &&  != 0 {
			if  != 0 {
				.w.Write(spaceBytes)
			}
			.w.Write(capEqualsBytes)
			printInt(.w, int64(), 10)
		}
		.w.Write(closeParenBytes)
		.w.Write(spaceBytes)
	}

	// Call Stringer/error interfaces if they exist and the handle methods flag
	// is enabled
	if !.cs.DisableMethods {
		if ( != reflect.Invalid) && ( != reflect.Interface) {
			if  := handleMethods(.cs, .w, );  {
				return
			}
		}
	}

	switch  {
	case reflect.Invalid:
		// Do nothing.  We should never get here since invalid has already
		// been handled above.

	case reflect.Bool:
		printBool(.w, .Bool())

	case reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64, reflect.Int:
		printInt(.w, .Int(), 10)

	case reflect.Uint8, reflect.Uint16, reflect.Uint32, reflect.Uint64, reflect.Uint:
		printUint(.w, .Uint(), 10)

	case reflect.Float32:
		printFloat(.w, .Float(), 32)

	case reflect.Float64:
		printFloat(.w, .Float(), 64)

	case reflect.Complex64:
		printComplex(.w, .Complex(), 32)

	case reflect.Complex128:
		printComplex(.w, .Complex(), 64)

	case reflect.Slice:
		if .IsNil() {
			.w.Write(nilAngleBytes)
			break
		}
		fallthrough

	case reflect.Array:
		.w.Write(openBraceNewlineBytes)
		.depth++

			.indent()
			.w.Write(maxNewlineBytes)
		} else {

		}

		.indent()
		.w.Write(closeBraceBytes)

	case reflect.String:

	case reflect.Interface:

		}

	case reflect.Ptr:

		// been handled above.

	case reflect.Map:

			break
		}

		.w.Write(closeBraceBytes)

		.depth++

		}
	}

/*
Dump displays the passed parameters to standard out with newlines, customizable
indentation, and additional debug information such as complete types and all
pointer addresses used to indirect to the final value.  It provides the
following features over the built-in printing facilities provided by the fmt
package:

	* Pointers are dereferenced and followed
	* Circular data structures are detected and handled properly
	* Custom Stringer/error interfaces are optionally invoked, including
	  on unexported types
	* Custom types which only implement the Stringer/error interfaces via
	  a pointer receiver are optionally invoked when passing non-pointer
	  variables
	* Byte arrays and slices are dumped like the hexdump -C command which
	  includes offsets, byte values in hex, and ASCII output

The configuration options are controlled by an exported package global,
spew.Config.  See ConfigState for options documentation.

See Fdump if you would prefer dumping to an arbitrary io.Writer or Sdump to
get the formatted result as a string.
*/
func ( ...interface{}) {
	fdump(&Config, os.Stdout, ...)
}
