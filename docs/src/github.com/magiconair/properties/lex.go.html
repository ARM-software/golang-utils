// item represents a token or text string returned from the scanner.
type item struct {
	typ itemType // The type of this item.
	pos int      // The starting position, in bytes, of this item in the input string.
	val string   // The value of this item.
}

func ( item) () string {
	switch {
	case .typ == itemEOF:
		return "EOF"
	case .typ == itemError:
		return .val
	case len(.val) > 10:
		return fmt.Sprintf("%.10q...", .val)
	}
	return fmt.Sprintf("%q", .val)
}

// itemType identifies the type of lex items.
type itemType int

const (
	itemError itemType = iota // error occurred; value is text of error
	itemEOF
	itemKey     // a key
	itemValue   // a value
	itemComment // a comment
)

// stateFn represents the state of the scanner as a function that returns the next state.
type stateFn func(*lexer) stateFn

// lexer holds the state of the scanner.

// lexBeforeKey scans until a key begins.
func ( *lexer) stateFn {
	switch  := .next(); {
	case isEOF():
		.emit(itemEOF)
		return nil

	case isEOL():
		.ignore()
		return 

	case isComment():
		return lexComment

	case isWhitespace():
		.ignore()
		return 

	default:
		.backup()
		return lexKey
	}
}

// lexComment scans a comment line. The comment character has already been scanned.
func ( *lexer) stateFn {
	.acceptRun(whitespace)
	.ignore()
	for {
		switch  := .next(); {
		case isEOF():
			.ignore()
			.emit(itemEOF)
			return nil
		case isEOL():
			.emit(itemComment)
			return lexBeforeKey
		default:
			.appendRune()
		}
	}
}

// lexKey scans the key up to a delimiter
func ( *lexer) stateFn {
	var  rune

:
	for {
		switch  = .next(); {

		case isEscape():
			 := .scanEscapeSequence()
			if  != nil {
				return .errorf(.Error())
			}

		case isEndOfKey():
			.backup()
			break 

		case isEOF():
			break 

		default:
			.appendRune()
		}
	}

	if len(.runes) > 0 {
		.emit(itemKey)
	}

	if isEOF() {
		.emit(itemEOF)
		return nil
	}

	return lexBeforeValue
}

// lexBeforeValue scans the delimiter between key and value.
// Leading and trailing whitespace is ignored.
// We expect to be just after the key.
func ( *lexer) stateFn {
	.acceptRun(whitespace)
	.accept(":=")
	.acceptRun(whitespace)
	.ignore()
	return lexValue
}

// lexValue scans text until the end of the line. We expect to be just after the delimiter.
func ( *lexer) stateFn {
	for {
		switch  := .next(); {
		case isEscape():
			if isEOL(.peek()) {
				.next()
				.acceptRun(whitespace)
			} else {
				 := .scanEscapeSequence()
				if  != nil {
					return .errorf(.Error())
				}
			}

		case isEOL():
			.emit(itemValue)
			.ignore()
			return lexBeforeKey

		case isEOF():
			.emit(itemValue)
			.emit(itemEOF)
			return nil

		default:
			.appendRune()
		}
	}
}

// scanEscapeSequence scans either one of the escaped characters
// or a unicode literal. We expect to be after the escape character.
func ( *lexer) () error {
	switch  := .next(); {

	case isEscapedCharacter():
		.appendRune(decodeEscapedCharacter())
		return nil

	case atUnicodeLiteral():
		return .scanUnicodeLiteral()

	case isEOF():
		return fmt.Errorf("premature EOF")

	// silently drop the escape character and append the rune as is
	default:
		.appendRune()
		return nil
	}
}
