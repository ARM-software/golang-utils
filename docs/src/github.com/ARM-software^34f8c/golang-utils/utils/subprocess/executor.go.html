// Subprocess describes what a subproccess is as well as any monitoring it may need.
type Subprocess struct {
	mu                deadlock.RWMutex
	isRunning         atomic.Bool
	command           *command
	processMonitoring *subprocessMonitoring
	messsaging        *subprocessMessaging
}

	return
}

// Setup sets up a sub-process i.e. defines the command cmd and the messages on start, success and failure.
func ( *Subprocess) ( context.Context,  logs.Loggers,  string, ,  string,  string,  ...string) ( error) {
	if .IsOn() {
		 = .Stop()
		if  != nil {
			return
		}
	}
	.mu.Lock()
	defer .mu.Unlock()
	.isRunning.Store(false)
	.processMonitoring = newSubprocessMonitoring()
	.command = newCommand(, , ...)
	.messsaging = newSubprocessMessaging(, , , , .command.GetPath())
	.reset()
	return .check()
}

func ( *Subprocess) () ( error) {
	// In GO, there is no reentrant locks and so following what is described there
	// https://groups.google.com/forum/#!msg/golang-nuts/XqW1qcuZgKg/Ui3nQkeLV80J
	if .command == nil {
		 = fmt.Errorf("missing command: %w", commonerrors.ErrUndefined)
		return
	}
	 = .command.Check()
	if  != nil {
		return
	}
	if .messsaging == nil {
		 = commonerrors.ErrNoLogger
		return
	}
	 = .messsaging.Check()
	return
}

// Check checks whether the subprocess is correctly defined.
func ( *Subprocess) () error {
	.mu.RLock()
	defer .mu.RUnlock()
	return .check()
}

// IsOn states whether the subprocess is running or not.
func ( *Subprocess) () bool {
	return .isRunning.Load() && .processMonitoring.IsOn()
}

// Start starts the process if not already started.
// This method is idempotent.
func ( *Subprocess) () ( error) {
	if .IsOn() {
		return
	}
	.mu.Lock()
	defer .mu.Unlock()
	 = .check()
	if  != nil {
		return
	}
	if .IsOn() {
		return
	}
	.reset()
	.runProcessMonitoring()
	 := .getCmd()
	 = .Start()
	if  != nil {
		.messsaging.LogFailedStart()
		.isRunning.Store(false)
		.Cancel()
		return
	}
	,  := .Pid()
	if  != nil {
		.messsaging.LogFailedStart()
		.isRunning.Store(false)
		.Cancel()
		return
	}

	.isRunning.Store(true)
	.messsaging.SetPid()
	.messsaging.LogStarted()
	return
}

// Cancel interrupts an on-going process. This method is idempotent.
func ( *Subprocess) () {
	.processMonitoring.CancelSubprocess()
}

// Execute executes the command and waits for completion.
func ( *Subprocess) () ( error) {
	 = .Check()
	if  != nil {
		return
	}
	.mu.Lock()
	defer .mu.Unlock()
	defer .Cancel()

	if .IsOn() {
		return fmt.Errorf("process is already started: %w", commonerrors.ErrConflict)
	}
	.processMonitoring.Reset()
	.command.Reset()
	.messsaging.LogStart()
	.runProcessMonitoring()
	 := .getCmd()
	.isRunning.Store(true)
	 = .Run()
	.isRunning.Store(false)
	.messsaging.LogEnd()
	return
}

// Stop stops the process straight away if currently working without waiting for completion. This method should be used in combination with `Start`.
// However, in order to interrupt a process however it was started (using `Start` or `Execute`), prefer `Cancel`.
// This method is idempotent.
func ( *Subprocess) () ( error) {
	return .stop(true)
}

// Restart restarts a process. It will stop the process if currently running.
func ( *Subprocess) () ( error) {
	 = .stop(false)
	if  != nil {
		return
	}
	return .Start()
}
func ( *Subprocess) () *cmdWrapper {
	return .command.GetCmd(.processMonitoring.ProcessContext())
}

func ( *Subprocess) () {
	.processMonitoring.RunMonitoring(.Stop)
}

func ( *Subprocess) () {
	.processMonitoring.Reset()
	.command.Reset()
}

func ( *Subprocess) ( bool) ( error) {
	if !.IsOn() {
		return
	}
	 = .Check()
	if  != nil {
		return
	}
	.mu.Lock()
	defer .mu.Unlock()
	defer func() {
		if  {
			.Cancel()
		}
	}()
	if !.IsOn() {
		return
	}
	.messsaging.LogStopping()
	 = .getCmd().Stop()
	.command.Reset()
	.isRunning.Store(false)
	.messsaging.LogEnd(nil)
	return
}
