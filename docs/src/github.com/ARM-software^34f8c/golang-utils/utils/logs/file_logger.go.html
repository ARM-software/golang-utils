type FileLoggers struct {
	GenericLoggers
	File *os.File
}

func ( *FileLoggers) () ( error) {
	 = .GenericLoggers.Check()
	if  != nil {
		return
	}
	if .File == nil {
		 = errors.New("undefined log file")
	}
	return
}

// Closes the logger
func ( *FileLoggers) () ( error) {
	if .File != nil {
		 = .File.Close()
	}
	if  != nil {
		return
	}
	 = .GenericLoggers.Close()
	return
}

		},

	}
	return
}
