// Checks whether the loggers are correctly defined or not.
func ( *JSONLoggers) () error {
	if .GetSource() == "" {
		return commonerrors.ErrNoLogSource
	}
	if .GetLoggerSource() == "" {
		return commonerrors.ErrNoLoggerSource
	}
	return nil
}

func ( *JSONLoggers) () error {
	zerolog.TimestampFieldName = "ctime"
	zerolog.MessageFieldName = "message"
	zerolog.LevelFieldName = "severity"
	.Zerologger = .Zerologger.With().Timestamp().Logger()
	return nil
}

// Logs to the output logger.
func ( *JSONLoggers) ( ...interface{}) {
	if len() == 1 && [0] == "\n" {
		return
	}
	.Zerologger.Info().Str("source", .GetLoggerSource()).Msg(fmt.Sprint(...))
}

// Logs to the Error logger.
func ( *JSONLoggers) ( ...interface{}) {
	if len() == 1 && [0] == "\n" {
		return
	}
	.Zerologger.Error().Str("source", .GetLoggerSource()).Msg(fmt.Sprint(...))
}

// Closes the logger
func ( *JSONLoggers) () error {
	.mu.Lock()
	defer .mu.Unlock()
	return .writer.Close()
}
