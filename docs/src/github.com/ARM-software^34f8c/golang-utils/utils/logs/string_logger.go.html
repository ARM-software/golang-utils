type StringWriter struct {
	io.WriteCloser
	mu   sync.RWMutex
	Logs strings.Builder
}

func ( *StringWriter) ( []byte) ( int,  error) {
	.mu.RLock()
	defer .mu.RUnlock()
	.Logs.Write()
	return
}

func ( *StringWriter) () ( error) {
	.mu.Lock()
	defer .mu.Unlock()
	.Logs.Reset()
	return
}

func ( *StringWriter) () string {
	.mu.Lock()
	defer .mu.Unlock()
	return .Logs.String()
}

type StringLoggers struct {
	GenericLoggers
	LogWriter StringWriter
}

func ( *StringLoggers) () error {
	return .GenericLoggers.Check()
}

func ( *StringLoggers) () string {
	return .LogWriter.GetFullContent()
}

// Closes the logger
func ( *StringLoggers) () ( error) {
	 = .LogWriter.Close()
	if  != nil {
		return
	}
	 = .GenericLoggers.Close()
	return
}

// Creates a logger to standard output/error
func ( string) ( *StringLoggers,  error) {
	 = &StringLoggers{
		LogWriter: StringWriter{},
	}
	(*).GenericLoggers = GenericLoggers{
		Output: log.New(&(*).LogWriter, fmt.Sprintf("[%v] Output: ", ), log.LstdFlags),
		Error:  log.New(&(*).LogWriter, fmt.Sprintf("[%v] Error: ", ), log.LstdFlags),
	}
	return
}
