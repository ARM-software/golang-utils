// Creates a logger to logrus logger (https://github.com/Sirupsen/logrus)

	 = &GenericLoggers{
		Output: log.New(.WriterLevel(logrus.InfoLevel), fmt.Sprintf("[%v] ", ), log.LstdFlags),

	}
	return
