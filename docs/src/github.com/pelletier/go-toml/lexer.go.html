// State functions

func ( *tomlLexer) () tomlLexStateFn {
	for {
		 := .peek()
		switch  {
		case '}': // after '{'
			return .lexRightCurlyBrace
		case '[':
			return .lexTableKey
		case '#':
			return .lexComment(.)
		case '=':
			return .lexEqual
		case '\r':
			fallthrough
		case '\n':
			.skip()
			continue
		}

		if isSpace() {
			.skip()
		}

		if isKeyStartChar() {
			return .lexKey
		}

		if  == eof {
			.next()
			break
		}
	}

	.emit(tokenEOF)
	return nil
}

func ( *tomlLexer) () tomlLexStateFn {
	for {
		 := .peek()
		switch  {
		case '.':
			return .errorf("cannot start float with a dot")
		case '=':
			return .lexEqual
		case '[':
			return .lexLeftBracket
		case ']':
			return .lexRightBracket
		case '{':
			return .lexLeftCurlyBrace
		case '}':
			return .lexRightCurlyBrace
		case '#':
			return .lexComment(.)
		case '"':
			return .lexString
		case '\'':
			return .lexLiteralString
		case ',':
			return .lexComma
		case '\r':
			fallthrough
		case '\n':
			.skip()
			if len(.brackets) > 0 && .brackets[len(.brackets)-1] == '[' {
				return .
			}
			return .lexVoid
		}

		if .follow("true") {
			return .lexTrue
		}

		if .follow("false") {
			return .lexFalse
		}

		if .follow("inf") {
			return .lexInf
		}

		if .follow("nan") {
			return .lexNan
		}

		if isSpace() {
			.skip()
			continue
		}

		if  == eof {
			.next()
			break
		}

		if  == '+' ||  == '-' {
			return .lexNumber
		}

		if isDigit() {
			return .lexDateTimeOrNumber
		}

		return .errorf("no value can start with %c", )
	}

	.emit(tokenEOF)
	return nil
}

func ( *tomlLexer) () tomlLexStateFn {
	// Could be either a date/time, or a digit.
	// The options for date/times are:
	//   YYYY-... => date or date-time
	//   HH:... => time
	// Anything else should be a number.

	 := .peekString(5)
	if len() < 3 {
		return .lexNumber()
	}

	for ,  := range  {
		if !isDigit() {
			if  == 2 &&  == ':' {
				return .lexDateTimeOrTime()
			}
			if  == 4 &&  == '-' {
				return .lexDateTimeOrTime()
			}
			return .lexNumber()
		}
	}
	return .lexNumber()
}

	// we already know those two are digits
	.next()

func ( *tomlLexer) () tomlLexStateFn {
	// This state accepts an offset date-time, a local date-time, or a local date.
	//
	//   v--- cursor
	// 1979-05-27T07:32:00Z
	// 1979-05-27T00:32:00-07:00
	// 1979-05-27T00:32:00.999999-07:00
	// 1979-05-27 07:32:00Z
	// 1979-05-27 00:32:00-07:00
	// 1979-05-27 00:32:00.999999-07:00
	// 1979-05-27T07:32:00
	// 1979-05-27T00:32:00.999999
	// 1979-05-27 07:32:00
	// 1979-05-27 00:32:00.999999
	// 1979-05-27

	// date

	// already checked by lexRvalue
	.next() // digit
	.next() // -

	for  := 0;  < 2; ++ {
		 := .next()
		if !isDigit() {
			return .errorf("invalid month digit in date: %c", )
		}
	}

	 := .next()
	if  != '-' {
		return .errorf("expected - to separate month of a date, not %c", )
	}

	for  := 0;  < 2; ++ {
		 := .next()
		if !isDigit() {
			return .errorf("invalid day digit in date: %c", )
		}
	}

	.emit(tokenLocalDate)

	 = .peek()

	if  == eof {

		return .lexRvalue
	}

	if  != ' ' &&  != 'T' {
		return .errorf("incorrect date/time separation character: %c", )
	}

	if  == ' ' {
		 := .peekString(3)[1:]
		if len() < 2 {
			return .lexRvalue
		}
		for ,  := range  {
			if !isDigit() {
				return .lexRvalue
			}
		}
	}

	.skip() // skip the T or ' '

	// time

	for  := 0;  < 2; ++ {
		 := .next()
		if !isDigit() {
			return .errorf("invalid hour digit in time: %c", )
		}
	}

	 = .next()
	if  != ':' {
		return .errorf("time hour/minute separator should be :, not %c", )
	}

	for  := 0;  < 2; ++ {
		 := .next()
		if !isDigit() {
			return .errorf("invalid minute digit in time: %c", )
		}
	}

	 = .next()
	if  != ':' {
		return .errorf("time minute/second separator should be :, not %c", )
	}

	for  := 0;  < 2; ++ {
		 := .next()
		if !isDigit() {
			return .errorf("invalid second digit in time: %c", )
		}
	}

	 = .peek()
	if  == '.' {
		.next()
		 := .next()
		if !isDigit() {
			return .errorf("expected at least one digit in time's fraction, not %c", )
		}

		for {
			 := .peek()
			if !isDigit() {
				break
			}
			.next()
		}
	}

	.emit(tokenLocalTime)

	return .lexTimeOffset

}

// Parse the key and emits its value without escape sequences.
// bare keys, basic string keys and literal string keys are supported.
func ( *tomlLexer) () tomlLexStateFn {
	var  strings.Builder

	for  := .peek(); isKeyChar() ||  == '\n' ||  == '\r';  = .peek() {
		if  == '"' {
			.next()
			,  := .lexStringAsString(`"`, false, true)
			if  != nil {
				return .errorf(.Error())
			}
			.WriteString("\"")
			.WriteString()
			.WriteString("\"")
			.next()
			continue
		} else if  == '\'' {
			.next()
			,  := .lexLiteralStringAsString(`'`, false)
			if  != nil {
				return .errorf(.Error())
			}
			.WriteString("'")
			.WriteString()
			.WriteString("'")
			.next()
			continue
		} else if  == '\n' {
			return .errorf("keys cannot contain new lines")
		} else if isSpace() {
			var  strings.Builder
			.WriteString(" ")

			// skip trailing whitespace
			.next()
			for  = .peek(); isSpace();  = .peek() {
				.WriteRune()
				.next()
			}
			// break loop if not a dot
			if  != '.' {
				break
			}
			.WriteString(".")
			// skip trailing whitespace after dot
			.next()
			for  = .peek(); isSpace();  = .peek() {
				.WriteRune()
				.next()
			}
			.WriteString(.String())
			continue
		} else if  == '.' {
			// skip
		} else if !isValidBareChar() {
			return .errorf("keys cannot contain %c character", )
		}
		.WriteRune()
		.next()
	}
	.emitWithValue(tokenKey, .String())
	return .lexVoid
}

// Lex a string and return the results as a string.
// Terminator is the substring indicating the end of the token.
// The resulting string does not include the terminator.
func ( *tomlLexer) ( string, ,  bool) (string, error) {
	var  strings.Builder

	if  {
		if .follow("\r\n") {
			.skip()
			.skip()
		} else if .peek() == '\n' {
			.skip()
		}
	}

	for {
		if .follow() {
			return .String(), nil
		}

		if .follow("\\") {
			.next()
			switch .peek() {
			case '\r':
				fallthrough
			case '\n':
				fallthrough
			case '\t':
				fallthrough
			case ' ':
				// skip all whitespace chars following backslash
				for strings.ContainsRune("\r\n\t ", .peek()) {
					.next()
				}
			case '"':
				.WriteString("\"")
				.next()
			case 'n':
				.WriteString("\n")
				.next()
			case 'b':
				.WriteString("\b")
				.next()
			case 'f':
				.WriteString("\f")
				.next()
			case '/':
				.WriteString("/")
				.next()
			case 't':
				.WriteString("\t")
				.next()
			case 'r':
				.WriteString("\r")
				.next()
			case '\\':
				.WriteString("\\")
				.next()
			case 'u':
				.next()
				var  strings.Builder
				for  := 0;  < 4; ++ {
					 := .peek()
					if !isHexDigit() {
						return "", errors.New("unfinished unicode escape")
					}
					.next()
					.WriteRune()
				}
				,  := strconv.ParseInt(.String(), 16, 32)
				if  != nil {
					return "", errors.New("invalid unicode escape: \\u" + .String())
				}
				.WriteRune(rune())
			case 'U':
				.next()
				var  strings.Builder
				for  := 0;  < 8; ++ {
					 := .peek()
					if !isHexDigit() {
						return "", errors.New("unfinished unicode escape")
					}
					.next()
					.WriteRune()
				}
				,  := strconv.ParseInt(.String(), 16, 64)
				if  != nil {
					return "", errors.New("invalid unicode escape: \\U" + .String())
				}
				.WriteRune(rune())
			default:
				return "", errors.New("invalid escape sequence: \\" + string(.peek()))
			}
		} else {
			 := .peek()

			if 0x00 <=  &&  <= 0x1F &&  != '\t' && !( && ( == '\n' ||  == '\r')) {
				return "", fmt.Errorf("unescaped control character %U", )
			}
			.next()
			.WriteRune()
		}

		if .peek() == eof {
			break
		}
	}

	return "", errors.New("unclosed string")
}

// Parse the key till "]]", but only bare keys are supported
func ( *tomlLexer) () tomlLexStateFn {
	for  := .peek();  != eof;  = .peek() {
		switch  {
		case ']':
			if .currentTokenStop > .currentTokenStart {
				.emit(tokenKeyGroupArray)
			}
			.next()
			if .peek() != ']' {
				break
			}
			.next()
			.emit(tokenDoubleRightBracket)
			return .lexVoid
		case '[':
			return .errorf("table array key cannot contain ']'")
		default:
			.next()
		}
	}
	return .errorf("unclosed table array key")
}

// Parse the key till "]" but only bare keys are supported

type validRuneFn func(r rune) bool

func ( rune) bool {
	return  >= 'a' &&  <= 'f' ||
		 >= 'A' &&  <= 'F' ||
		 >= '0' &&  <= '9' ||
		 == '_'
}

func ( rune) bool {
	return  >= '0' &&  <= '7' ||  == '_'
}

func ( rune) bool {
	return  == '0' ||  == '1' ||  == '_'
}
