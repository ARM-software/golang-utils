// Close all poller file descriptors, but not the one passed to it.
func ( *fdPoller) () {
	if .pipe[1] != -1 {
		unix.Close(.pipe[1])
	}
	if .pipe[0] != -1 {
		unix.Close(.pipe[0])
	}
	if .epfd != -1 {
		unix.Close(.epfd)
	}
}
