// Token defines a single HCL token which can be obtained via the Scanner
type Token struct {
	Type Type
	Pos  Pos
	Text string
}

// Type is the set of lexical tokens of the HCL (HashiCorp Configuration Language)
type Type int

const (
	// Special tokens
	ILLEGAL Type = iota
	EOF

	identifier_beg
	literal_beg
	NUMBER // 12345
	FLOAT  // 123.45
	BOOL   // true,false
	STRING // "abc"
	NULL   // null
	literal_end
	identifier_end

	operator_beg
	LBRACK // [
	LBRACE // {
	COMMA  // ,
	PERIOD // .
	COLON  // :

	RBRACK // ]
	RBRACE // }

	operator_end
)

var tokens = [...]string{

// String returns the token's literal text. Note that this is only
// applicable for certain token types, such as token.IDENT,
// token.STRING, etc..
func ( Token) () string {
	return fmt.Sprintf("%s %s %s", .Pos.String(), .Type.String(), .Text)
}

// HCLToken converts this token to an HCL token.
//
// The token type must be a literal type or this will panic.
func ( Token) () hcltoken.Token {
	switch .Type {
	case BOOL:
		return hcltoken.Token{Type: hcltoken.BOOL, Text: .Text}
	case FLOAT:
		return hcltoken.Token{Type: hcltoken.FLOAT, Text: .Text}
	case NULL:
		return hcltoken.Token{Type: hcltoken.STRING, Text: ""}
	case NUMBER:
		return hcltoken.Token{Type: hcltoken.NUMBER, Text: .Text}
	case STRING:
		return hcltoken.Token{Type: hcltoken.STRING, Text: .Text, JSON: true}
	default:
		panic(fmt.Sprintf("unimplemented HCLToken for type: %s", .Type))
	}
}
