// Token defines a single HCL token which can be obtained via the Scanner
type Token struct {
	Type Type
	Pos  Pos
	Text string
	JSON bool

// Type is the set of lexical tokens of the HCL (HashiCorp Configuration Language)
type Type int

const (
	// Special tokens
	ILLEGAL Type = iota
	EOF
	COMMENT

	identifier_beg
	IDENT // literals
	literal_beg
	NUMBER  // 12345
	FLOAT   // 123.45
	BOOL    // true,false
	STRING  // "abc"
	HEREDOC // <<FOO\nbar\nFOO
	literal_end
	identifier_end

	operator_beg
	LBRACK // [
	LBRACE // {
	COMMA  // ,
	PERIOD // .

	RBRACK // ]
	RBRACE // }

	ASSIGN // =
	ADD    // +
	SUB    // -
	operator_end
)

// String returns the token's literal text. Note that this is only
// applicable for certain token types, such as token.IDENT,
// token.STRING, etc..
func ( Token) () string {
	return fmt.Sprintf("%s %s %s", .Pos.String(), .Type.String(), .Text)
}

// Value returns the properly typed value for this token. The type of
// the returned interface{} is guaranteed based on the Type field.
//
// This can only be called for literal types. If it is called for any other
// type, this will panic.
func ( Token) () interface{} {
	switch .Type {
	case BOOL:
		if .Text == "true" {
			return true
		} else if .Text == "false" {
			return false
		}

		panic("unknown bool value: " + .Text)
	case FLOAT:
		,  := strconv.ParseFloat(.Text, 64)
		if  != nil {
			panic()
		}

		return float64()
	case NUMBER:
		,  := strconv.ParseInt(.Text, 0, 64)
		if  != nil {
			panic()
		}

		return int64()
	case IDENT:
		return .Text
	case HEREDOC:
		return unindentHeredoc(.Text)
	case STRING:
		// Determine the Unquote method to use. If it came from JSON,
		// then we need to use the built-in unquote since we have to
		// escape interpolations there.
		 := hclstrconv.Unquote
		if .JSON {
			 = strconv.Unquote
		}

		// This case occurs if json null is used
		if .Text == "" {
			return ""
		}

		,  := (.Text)
		if  != nil {
			panic(fmt.Sprintf("unquote %s err: %s", .Text, ))
		}

		return 
	default:
		panic(fmt.Sprintf("unimplemented Value for type: %s", .Type))
	}
}
