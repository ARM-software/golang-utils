type CancelFunctionStore struct {
	mu              sync.RWMutex
	cancelFunctions []context.CancelFunc
}

func ( *CancelFunctionStore) ( ...context.CancelFunc) {
	defer .mu.Unlock()
	.mu.Lock()
	.cancelFunctions = append(.cancelFunctions, ...)
}

func ( *CancelFunctionStore) () {
	defer .mu.RUnlock()
	.mu.RLock()
	for ,  := range .cancelFunctions {
		()
	}
}

func ( *CancelFunctionStore) () int {
	defer .mu.RUnlock()
	.mu.RLock()
	return len(.cancelFunctions)
}

func () *CancelFunctionStore {
	return &CancelFunctionStore{
		cancelFunctions: []context.CancelFunc{},
	}
}
