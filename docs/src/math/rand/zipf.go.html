// A Zipf generates Zipf distributed variates.
type Zipf struct {
	r            *Rand
	imax         float64
	v            float64
	q            float64
	s            float64
	oneminusQ    float64
	oneminusQinv float64
	hxm          float64
	hx0minusHxm  float64
}

func ( *Zipf) ( float64) float64 {
	return math.Exp(.oneminusQ*math.Log(.v+)) * .oneminusQinv
}

func ( *Zipf) ( float64) float64 {
	return math.Exp(.oneminusQinv*math.Log(.oneminusQ*)) - .v
}
