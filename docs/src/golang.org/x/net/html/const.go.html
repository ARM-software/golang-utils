func isSpecialElement( *Node) bool {
	switch .Namespace {
	case "", "html":
		return isSpecialElementMap[.Data]
	case "math":
		switch .Data {
		case "mi", "mo", "mn", "ms", "mtext", "annotation-xml":
			return true
		}
	case "svg":
		switch .Data {
		case "foreignObject", "desc", "title":
			return true
		}
	}
	return false
}
