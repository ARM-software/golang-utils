package unix

// Set adds fd to the set fds.
func ( *FdSet) ( int) {
	.Bits[/NFDBITS] |= (1 << (uintptr() % NFDBITS))
}

// Clear removes fd from the set fds.
func ( *FdSet) ( int) {
	.Bits[/NFDBITS] &^= (1 << (uintptr() % NFDBITS))
}

// IsSet returns whether fd is in the set fds.
func ( *FdSet) ( int) bool {
	return .Bits[/NFDBITS]&(1<<(uintptr()%NFDBITS)) != 0
}

// Zero clears the set fds.
func ( *FdSet) () {
	for  := range .Bits {
		.Bits[] = 0
	}
}
