// An extFloat represents an extended floating-point number, with more
// precision than a float64. It does not try to save bits: the
// number represented by the structure is mant*(2^exp), with a negative
// sign if neg is true.

	mant uint64
	exp  int
	neg  bool
}

// (den << shift)/2.
